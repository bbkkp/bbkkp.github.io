OBJS=main.o tool1.o tool2.o
CC=gcc
CFLAGS=-c -Wall -g
mytool:$(OBJS)
    gcc  $(OBJS) -o mytool
main.o:main.c
    gcc main.c $(CFLAGS) -o main.o
tool1.o:tool1.c
    gcc tool1.c $(CFLAGS) -o tool1.o
tool2.o:tool2.c
    gcc tool2.c $(CFLAGS) -o tool2.o

    rm *.o mytool -rf

Makefile文件再优化

OBJS=main.o tool1.o tool2.o
CC=gcc
CFLAGS=-c -Wall -g
mytool:$(OBJS)
    gcc $^ -o $@
%.o:%.c
    gcc $^ $(CFLAGS) -o $@
clean:
    rm *.o mytool -rf
# $^代替依赖文件
# $@代替目标文件
# %.o 通配所有.o文件
# %.c通配所有.c文件
#查找main.o文件，发现没有。向下查找，到第6行自动匹配。

再优化，让makefile自己寻找.c文件对应的.o文件

OBJS= $(patsubst %.c,%.o,$(wildcard *.c))
CC=gcc
CFLAGS=-c -Wall -g
mytool:$(OBJS)
    gcc $^ -o $@
%.o:%.c
    gcc $^ $(CFLAGS) -o $@
clean:
    rm *.o mytool -rf

优化，解决.c文件依赖其他.h文件的问题

我们引入.d文件，这个文件的原理其实就是利用gcc的参数-MM，分析.c文件的内容，识别.c文件的头文件依赖，再通过sed命令将依赖写入.d文件

# 列出所有 .c 文件:
SRCS = $(wildcard *.c)

# 根据SRCS生成 .o 文件列表:
OBJS = $(SRCS:.c=.o)

# 根据SRCS生成 .d 文件列表:
DEPS = $(SRCS:.c=.d)

TARGET = mytool

# 默认目标:
$(TARGET): $(OBJS)
    $(CC) -o $@ $^

# xyz.d 的规则由 xyz.c 生成:
%.d: %.c
    rm -f $@; \
    $(CC) -MM $< >$@.tmp; \
    sed 's,\($*\)\.o[ :]*,\1.o $@ : ,g' < $@.tmp > $@; \
    rm -f $@.tmp

# 模式规则:
%.o: %.c
    $(CC) -c -o $@ $<

clean:
    rm -rf *.o *.d $(TARGET)

# 引入所有 .d 文件:
include $(DEPS)

优化，可以用，但是存在.d文件不存在时打印错误的问题(最终版)

项目结构

这个Makefile定义了源码目录SRC_DIR、生成目录BUILD_DIR，以及其他变量，同时用-include消除了.d文件不存在的错误。

后记

我曾经见过这样一种makefile,它在一开始就手动引入的所有文件，然后除了clean，两行代码搞定，理论上的确搞定了一个make解决一切的问题，但是再真实的项目中如果你这么写，会被吊的。除了一个make解决一切，更重要的一下规则

  1. 如果这个工程没有编译过，那么我们的所有c文件都要编译并被链接。
  2. 如果这个工程的某几个c文件被修改，那么我们只编译被修改的c文件，并链接目标程序。
  3. 如果这个工程的头文件被改变了，那么我们需要编译引用了这几个头文件的c文件，并链接目标程序。
